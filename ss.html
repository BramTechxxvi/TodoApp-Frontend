// Replace BASE_URL with your Spring Boot backend base path const BASE_URL = 'http://localhost:8080'; const USER_ID = localStorage.getItem('userId'); // assuming user ID is stored in localStorage const taskList = document.getElementById('taskList'); const addInput = document.getElementById('add-task'); window.onload = () => { loadUserEmail(); fetchTasks(); }; function loadUserEmail() { const email = localStorage.getItem('userEmail'); if (email) document.getElementById('user-email').textContent = `Welcome, ${email}`; } function changeEmail() { const email = document.getElementById('new-email').value; fetch(`${BASE_URL}/api/user/changeEmail`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ userId: USER_ID, newEmail: email }) }).then(() => { localStorage.setItem('userEmail', email); loadUserEmail(); alert('Email updated.'); }); } function changePassword() { const password = document.getElementById('new-password').value; fetch(`${BASE_URL}/api/user/changePassword`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ userId: USER_ID, newPassword: password }) }).then(() => alert('Password updated.')); } function renderTask(task) { const li = document.createElement('li'); li.dataset.id = task.id; li.innerHTML = ` ${task.title} ${task.status === 'completed' ? '✅ Completed' : '⏳ In Progress'} `; taskList.appendChild(li); } function updateStatus(id, status) { fetch(`${BASE_URL}/api/task/updateStatus/${id}`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ status }) }) .then(() => fetchTasks()); }
