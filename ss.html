window.onload = () => { loadUserEmail(); fetchTasks(); }; function loadUserEmail() { const email = localStorage.getItem('userEmail'); if (email) document.getElementById('user-email').textContent = `Welcome, ${email}`; } function changeEmail() { const email = document.getElementById('new-email').value; fetch(`${BASE_URL}/api/user/changeEmail`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ userId: USER_ID, newEmail: email }) }).then(() => { localStorage.setItem('userEmail', email); loadUserEmail(); alert('Email updated.'); }); } function changePassword() { const password = document.getElementById('new-password').value; fetch(`${BASE_URL}/api/user/changePassword`, { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ userId: USER_ID, newPassword: password }) }).then(() => alert('Password updated.')); }
